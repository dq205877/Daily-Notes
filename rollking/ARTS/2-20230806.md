```java
public int uniquePaths(int m, int n) {
        int[][] ans = new int[m][n];
        //先填满最后一行与最后一列
        for(int i = 0; i < m; i++){
            ans[i][0] = 1;
        }
        for(int j = 0; j < n; j++){
            ans[0][j] = 1;
        }
        //填充上面的格子数据
        for(int i = 1; i < m; i++){
            for(int j = 1; j < n; j++){
                ans[i][j] = ans[i - 1][j] + ans[i][j - 1];
            }
        }
        //返回结果
        return ans[m - 1][n - 1];

        // }
        // return (int) ans;
    }
```
